CREATE TABLESPACE FITZ_db
  DATAFILE 'FITZ_db.dat' 
    SIZE 20M
  ONLINE;

CREATE TEMPORARY TABLESPACE FITZ_db_tmp
  TEMPFILE 'FITZ_db_tmp1.dbf'
    SIZE 5M
    AUTOEXTEND ON;
    
CREATE USER FITZ
  IDENTIFIED BY FITZ_secret_password
  DEFAULT TABLESPACE FITZ_db
  TEMPORARY TABLESPACE FITZ_db_tmp
  QUOTA 20M on FITZ_db;    
    
-- ROLES
GRANT "DBA" TO FITZ WITH ADMIN OPTION;
